double tichle(int arr[][MAX] , int& n  , int&m){

	for(int i = 0 ; i < n ; i++){
		for(int j = 0 ; j < i ; j++){
			double tich = 1;
			cout<<arr[i][j]<<" ";
			if(arr[i][j] % 2 != 0){
				kq = tich *arr[i][j]; 
			}
			if(arr[i][j] < 0 && arr[i][j] % 2 != 0){
				cout<<"ko co phan tu le duoi duong cheo phu "<<endl;
				break;
			}
		}
		cout<<endl;
	}
	cout<<"tich cac diem nam duoi duong cheo phu la : "<<kq<<endl;

}
void xuat(int arr[][MAX] , int n , int m){

		for(int j = 0 ; j < m ; j++){
			cout<<arr[i][j]<<" ";
		}
		cout<<endl;
	}
}
